CREATE FUNCTION [dbo].[DAYSADDNOWK](@toDay AS DATE, @numDays AS INT)
RETURNS DATETIME
AS
BEGIN
DECLARE @addDate as DATE;
  --  SET @addDate = DATEADD(d, @numDays, @addDate)
    IF DATENAME(DW, @toDay) = 'Sunday'          SET @addDate = DATEADD(d, 3, @toDay)
    IF DATENAME(DW, @toDay) = 'Monday'          SET @addDate = DATEADD(d, 2, @toDay)
       IF DATENAME(DW, @toDay) = 'Tuesday'             SET @addDate = DATEADD(d, 2, @toDay)
      IF DATENAME(DW, @toDay) = 'Wednesday'   SET @addDate = DATEADD(d, 2, @toDay)
      IF DATENAME(DW, @toDay) = 'Thursday'     SET @addDate = DATEADD(d, 4, @toDay)
      IF DATENAME(DW, @toDay) = 'Friday'       SET @addDate = DATEADD(d, 4, @toDay)
      IF DATENAME(DW, @toDay) = 'Saturday'     SET @addDate = DATEADD(d, 3, @toDay)
 
    RETURN CAST(@addDate AS DATETIME)
END
 


CREATE FUNCTION DAYSADDNOWK(@addDate AS DATE, @numDays AS INT)
RETURNS DATETIME
AS
BEGIN
    SET @addDate = DATEADD(d, @numDays, @addDate)
    IF DATENAME(DW, @addDate) = 'Sunday'   SET @addDate = DATEADD(d, 1, @addDate)
    IF DATENAME(DW, @addDate) = 'Saturday' SET @addDate = DATEADD(d, 2, @addDate)
 
    RETURN CAST(@addDate AS DATETIME)
END
